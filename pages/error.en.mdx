import { TableData } from '@/components/errorTable';

export const getStaticProps = ({ params }) => {
  return fetch(
    `https://raw.githubusercontent.com/tgsnake/core/master/generator/error/source/errors.json`,
  )
    .then((res) => res.json())
    .then((list) => ({
      props: {
        ssg: {
          list: list,
        },
      },
      revalidate: 60,
    }));
};
